  #include <avr/power.h>
#endif

#define DELAYVAL 500

boolean etatLed = false;

  pinMode(A0, OUTPUT);
  Serial.begin(115000);
  #if defined(__AVR_ATtiny85__) && (F_CPU == 16000000)
  clock_prescale_set(clock_div_1);
#endif

  pixels.begin();
   srand(time(NULL));
}

void loop() {
  pixels.clear();
  pixels.show();
  
}
